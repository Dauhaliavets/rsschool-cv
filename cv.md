const getBattleshipCount = board => {
    let count = 0;
    for (let i = 0; i < board.length; i++) {
        for (let j = 0; j < board[i].length; j++) {
            if (board[i][j] === '.') continue;
            if (i > 0 && board[i - 1][j] === 'X') continue;

    }
